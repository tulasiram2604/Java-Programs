//By using decrement 

1.begin
2.declare num,result,count
3.read num and initialize count=num and result=1
4.while(count>=1)do
5.result=result*count
6.count=count-1
7.end while
8.display result
9.end


//By using increment

1.begin
2.declare num,count,result
3.read num and initialize result=1 and count=1
4.while(count<=num)
5.result=result*count
6.count=count+1
7.end while
8.print result
9.end
